\documentclass[border=5mm,tikz]{standalone}
\usepackage{pgfplots}
\pgfplotsset{compat=1.8}

\usepackage{amsmath}
\pgfplotsset{ % Define a common style, so we don't repeat ourselves
    MaoYiyi/.style={
        width=0.6\textwidth, % Overall width of the plot
        axis equal image, % Unit vectors for both axes have the same length
        view={0}{90}, % We need to use "3D" plots, but we set the view so we look at them from straight up
        xmin=0, xmax=10, % Axis limits
        ymin=0, ymax=10,
        domain=0:10, y domain=0:10, % Domain over which to evaluate the functions
        xtick={0,1,...,10}, 
        ytick={0,1,...,10}, % Tick marks
        samples=11, % How many arrows?
        cycle list={    % Plot styles
                gray,
                quiver={
                    u={1}, v={f(x)}, % End points of the arrows
                    scale arrows=0.75,
                    every arrow/.append style={
                        -latex[white] % Arrow tip
                    },
                }\\
                red, samples=31, smooth, thick, no markers, domain=0:11\\ % The plot style for the function
        }
    }
}

\begin{document}
\begin{tikzpicture}[
    declare function={f(\x) = \y;}
]
\begin{axis}[
    MaoYiyi,
    title={$\dfrac{\mathrm{d}y}{\mathrm{d}x}=y$},
    ytick=\empty
]
\addplot3 (x,y,0);
\addplot +[domain=0.001:11] {e^(x-1)};
\end{axis}
\end{tikzpicture}
\end{document}
